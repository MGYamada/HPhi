75 int CalcSpectrum(

89  double complex OmegaMax, OmegaMin;

91  double complex *dcomega;
92 
93  //set omega
94  if(SetOmega(&(X->Bind.Def)) != TRUE){
95  fprintf(stderr, "Error: Fail to set Omega.\n");
96  exitMPI(-1);
97  }
98  else{
99  if(X->Bind.Def.iFlgSpecOmegaIm == FALSE){
100  X->Bind.Def.dOmegaIm = (X->Bind.Def.dOmegaMax - X->Bind.Def.dOmegaMin)/(double) X->Bind.Def.iNOmega;
101  }
102  }
103  /*
104  Set & malloc omega grid
105  */
106  Nomega = X->Bind.Def.iNOmega;
107  c_malloc1(dcSpectrum, Nomega);
108  c_malloc1(dcomega, Nomega);
109  OmegaMax = X->Bind.Def.dOmegaMax + X->Bind.Def.dOmegaIm*I;
110  OmegaMin = X->Bind.Def.dOmegaMin + X->Bind.Def.dOmegaIm*I;
111  for (i = 0; i< Nomega; i++) {
112  dcomega[i] = (OmegaMax - OmegaMin) / Nomega*i + OmegaMin;
113  }
114 
115  if(X->Bind.Def.NSingleExcitationOperator == 0 && X->Bind.Def.NPairExcitationOperator == 0){
116  fprintf(stderr, "Error: Any excitation operators are not defined.\n");
117  exitMPI(-1);
118  }
119 
120  //Make excited state
121  StartTimer(6100);
122  if (X->Bind.Def.iFlgCalcSpec == RECALC_NOT ||
123  X->Bind.Def.iFlgCalcSpec == RECALC_OUTPUT_TMComponents_VEC) {
124  //input eigen vector
125  StartTimer(6101);
126  fprintf(stdoutMPI, " Start: An Eigenvector is inputted in CalcSpectrum.\n");
127  TimeKeeper(&(X->Bind), cFileNameTimeKeep, c_InputEigenVectorStart, "a");
128  GetFileNameByKW(KWSpectrumVec, &defname);
129  strcat(defname, "_rank_%d.dat");
130 // sprintf(sdt, cFileNameInputEigen, X->Bind.Def.CDataFileHead, X->Bind.Def.k_exct - 1, myrank);
131  sprintf(sdt, defname, myrank);
132  childfopenALL(sdt, "rb", &fp);
133  if (fp == NULL) {
134  fprintf(stderr, "Error: A file of Inputvector does not exist.\n");
135  return -1;
136  }
137  fread(&i_stp, sizeof(i_stp), 1, fp);
138  X->Bind.Large.itr=i_stp; //For TPQ
139  fread(&i_max, sizeof(i_max), 1, fp);
140  if (i_max != X->Bind.Check.idim_max) {
141  fprintf(stderr, "Error: myrank=%d, i_max=%ld\n", myrank, i_max);
142  fprintf(stderr, "Error: A file of Inputvector is incorrect.\n");
143  return -1;
144  }
145 
146  fread(v1, sizeof(complex double), X->Bind.Check.idim_max + 1, fp);
147  fclose(fp);
148  StopTimer(6101);
149 
150  for (i = 1; i <= i_max; i++) {
151  v0[i] = 0;
152  }
153  fprintf(stdoutMPI, " End: An Inputcector is inputted in CalcSpectrum.\n\n");
154  TimeKeeper(&(X->Bind), cFileNameTimeKeep, c_InputEigenVectorEnd, "a");
155  TimeKeeper(&(X->Bind), cFileNameTimeKeep, c_CalcExcitedStateStart, "a");
156  fprintf(stdoutMPI, " Start: Calculating an excited Eigenvector.\n");
157  //mltply Operator
158  StartTimer(6102);
159  GetExcitedState(&(X->Bind), v0, v1);
160  StopTimer(6102);
161  //calculate norm
162  dnorm = NormMPI_dc(i_max, v0);
163  if (fabs(dnorm) < pow(10.0, -15)) {
164  fprintf(stderr, "Error: Excitation vector is illegal; norm becomes 0.\n");
165  return -1;
166  }
167  //normalize vector
168 #pragma omp parallel for default(none) private(i) shared(v1, v0) firstprivate(i_max, dnorm)
169  for (i = 1; i <= i_max; i++) {
170  v1[i] = v0[i] / dnorm;
171  }
172  fprintf(stdoutMPI, " End: Calculating an excited Eigenvector.\n\n");
173  TimeKeeper(&(X->Bind), cFileNameTimeKeep, c_CalcExcitedStateEnd, "a");
174  }
175  StopTimer(6100);
176 
177 
178  int iret=TRUE;
179  fprintf(stdoutMPI, " Start: Caclulating a spectrum.\n\n");
180  TimeKeeper(&(X->Bind), cFileNameTimeKeep, c_CalcSpectrumStart, "a");
181  StartTimer(6200);
182  switch (X->Bind.Def.iCalcType) {
183  case Lanczos:
184  iret = CalcSpectrumByLanczos(X, v1, dnorm, Nomega, dcSpectrum, dcomega);
185  if (iret != TRUE) {
186  //Error Message will be added.
187  return FALSE;
188  }
189 
190  break;//Lanczos Spectrum
191 
192  case TPQCalc:
193  fprintf(stderr, " Error: TPQ is not supported for calculating spectrum.\n");
194  return FALSE;//TPQ is not supprted.
195  iret = CalcSpectrumByTPQ(X, v1, dnorm, Nomega, dcSpectrum, dcomega);
196  if (iret != TRUE) {
197  //Error Message will be added.
198  return FALSE;
199  }
200  break;
201 
202  case FullDiag:
203  iret = CalcSpectrumByFullDiag(X, Nomega, dcSpectrum, dcomega);
204  break;
205 
206  // case CalcSpecByShiftedKlyrov will be added
207  default:
208  break;
209  }
210  StopTimer(6200);
211 
212  if (iret != TRUE) {
213  //Error Message will be added.
214  return FALSE;
215  }
216  else {
217  fprintf(stdoutMPI, " End: Calculating a spectrum.\n\n");
218  TimeKeeper(&(X->Bind), cFileNameTimeKeep, c_CalcSpectrumEnd, "a");
219  iret = OutputSpectrum(X, Nomega, dcSpectrum, dcomega);
220  return TRUE;
221  }
