72  for(isite2=1;isite2<=X->Def.Nsite;isite2++){
73  is1_up=X->Def.Tpow[2*isite1-2];
74  is1_down=X->Def.Tpow[2*isite1-1];

79 #pragma omp parallel for reduction(+:spn) default(none) firstprivate(i_max, is1_up, is1_down, is2_up, is2_down, irght, ilft, ihfbit, isite1, isite2) private(ibit1_up, num1_up, ibit2_up, num2_up, ibit1_down, num1_down, ibit2_down, num2_down, spn_z, iexchg, off) shared(vec, list_1, list_2_1, list_2_2)

81 
82  ibit1_up= list_1[j]&is1_up;
83  num1_up=ibit1_up/is1_up;
84  ibit2_up= list_1[j]&is2_up;
85  num2_up=ibit2_up/is2_up;
86 
87  ibit1_down= list_1[j]&is1_down;
88  num1_down=ibit1_down/is1_down;
89  ibit2_down= list_1[j]&is2_down;
90  num2_down=ibit2_down/is2_down;
91 
92  spn_z=(num1_up-num1_down)*(num2_up-num2_down);
93  spn+= conj(vec[j])*vec[j]*spn_z/4.0;
94  if(isite1==isite2){
95  spn+=conj(vec[j])*vec[j]*(num1_up+num1_down-2*num1_up*num1_down)/2.0;
96  }else{
97  if(ibit1_up!=0 && ibit1_down==0 && ibit2_up==0 &&ibit2_down!=0 ){
98  iexchg= list_1[j]-(is1_up+is2_down);
99  iexchg+=(is2_up+is1_down);
100  GetOffComp(list_2_1, list_2_2, iexchg, irght, ilft, ihfbit, &off);
101  spn+=conj(vec[j])*vec[off]/2.0;
102  }else if(ibit1_up==0 && ibit1_down!=0 && ibit2_up!=0 && ibit2_down==0){
103  iexchg= list_1[j]-(is1_down+is2_up);
104  iexchg+=(is2_down+is1_up);
105  GetOffComp(list_2_1, list_2_2, iexchg, irght, ilft, ihfbit, &off);

110  }
