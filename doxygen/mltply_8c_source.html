38  long unsigned int j=0;
39  long unsigned int irght=0;
40  long unsigned int ilft=0;
41  long unsigned int ihfbit=0;
42 
43  double complex dam_pr;
44 
45  long unsigned int i_max;
46 
47  StartTimer(1);
48  i_max = X->Check.idim_max;
49  X->Large.prdct = 0.0;
50  dam_pr = 0.0;
51 
52  if(i_max!=0){
53  if (X->Def.iFlgGeneralSpin == FALSE) {
54  if (GetSplitBitByModel(X->Def.Nsite, X->Def.iCalcModel, &irght, &ilft, &ihfbit) != 0) {
55  return -1;
56  }
57  }
58  else{
59  if(X->Def.iCalcModel==Spin){

61  return -1;
62  }
63  }
64  }
65  }
66  else{
67  irght=0;
68  ilft=0;
69  ihfbit=0;
70  }
71  X->Large.i_max = i_max;
72  X->Large.irght = irght;
73  X->Large.ilft = ilft;
74  X->Large.ihfbit = ihfbit;
75  X->Large.mode = M_MLTPLY;
76 
77  StartTimer(100);
78 #pragma omp parallel for default(none) reduction(+:dam_pr) firstprivate(i_max) shared(tmp_v0, tmp_v1, list_Diagonal)
79  for (j = 1; j <= i_max; j++) {
80  tmp_v0[j] += (list_Diagonal[j]) * tmp_v1[j];
81  dam_pr += (list_Diagonal[j]) * conj(tmp_v1[j]) * tmp_v1[j];
82  }
83  X->Large.prdct += dam_pr;
84  StopTimer(100);
85 
86  switch (X->Def.iCalcModel) {
87  case HubbardGC:
88  mltplyHubbardGC(X, tmp_v0, tmp_v1);
89  break;
90 
91  case KondoGC:
92  case Hubbard:
93  case Kondo:
94  mltplyHubbard(X, tmp_v0, tmp_v1);
95  break;
96 
97  case Spin:
98  mltplySpin(X, tmp_v0, tmp_v1);
99  break;
100 
101  case SpinGC:
102  mltplySpinGC(X, tmp_v0, tmp_v1);
103  break;
104 
105  default:
106  return -1;
107  }
108 
109  X->Large.prdct = SumMPI_dc(X->Large.prdct);
110  StopTimer(1);
111  return 0;
