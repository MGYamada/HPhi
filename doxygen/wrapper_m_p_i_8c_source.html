33 void InitializeMPI(int argc, char *argv[]){

35 
36 #ifdef MPI
37  ierr = MPI_Init(&argc, &argv);
38  ierr = MPI_Comm_size(MPI_COMM_WORLD, &nproc);
39  ierr = MPI_Comm_rank(MPI_COMM_WORLD, &myrank);
40  if(ierr != 0) exitMPI(ierr);
41 #else
42  nproc = 1;
43  myrank = 0;
44 #endif

47 
48 #pragma omp parallel default(none) shared(nthreads)
49 #pragma omp master
50  nthreads = omp_get_num_threads();
51 
52  expon = (int)(log((double)nproc) / log(2.0) + 0.5);
53 
54  fprintf(stdoutMPI, "\n\n##### Parallelization Info. #####\n\n");
55  fprintf(stdoutMPI, " OpenMP threads : %d\n", nthreads);
56  fprintf(stdoutMPI, " MPI PEs : %d = 2^%-5d\n\n", nproc, expon);
57  if (nproc != 1 << expon){
58  fprintf(stderr, "ERROR ! The number of PEs is not a 2-exponent !");
59  exitMPI(-1);
60  }
61 }
