177 int GetSpectrumByTridiagonalMatrixComponents(
178  double *tmp_alpha,
179  double *tmp_beta,
180  double dnorm,
181  double complex dcomega,
182  double complex *dcSpectrum,
183  unsigned long int ilLanczosStp
184  )
185 {
186  unsigned long int istp=2;
187  double complex dcDn;
188  double complex dcb0;
189  double complex dcbn, dcan;
190  double complex dcDeltahn;
191  double complex dch;
192 
193  if(ilLanczosStp < 1){
194  //TODO: Add error message
195  return FALSE;
196  }
197  dcb0 = dcomega-tmp_alpha[1];
198  if(ilLanczosStp ==1) {
199  if(cabs(dcb0)<eps_Energy){
200  dcb0=eps_Energy;
201  }
202  *dcSpectrum = dnorm / (dcb0);
203  return TRUE;
204  }
205 
206  dcbn = dcomega-tmp_alpha[2];
207  dcan = -pow(tmp_beta[1],2);
208  dcDn=1.0/dcbn;
209  dcDeltahn = dcan*dcDn;
210  dch=dcb0+dcDeltahn;
211 
212  for(istp=2; istp<=ilLanczosStp; istp++){
213  dcbn = dcomega-tmp_alpha[istp+1];
214  dcan =-pow(tmp_beta[istp],2);
215  dcDn = (dcbn+dcan*dcDn);
216  if(cabs(dcDn)<eps_Energy){
217  dcDn=eps_Energy;
218  }
219  dcDn =1.0/dcDn;
220  dcDeltahn = (dcbn*dcDn-1.0)*dcDeltahn;
221  dch += dcDeltahn;
222  if(cabs(dcDeltahn/dch)<cabs(dcb0)*eps) break;
223  }
224  *dcSpectrum = dnorm/(dch);
225  return TRUE;
226 }
